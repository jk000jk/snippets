class Stiva {
  int *v;	
  int dim; // dim max
  int crt; // nr crt de elem
  
  public: 
	Stiva(int dim);
	void push(int x);
	int pop();	
	void afisare();
};

# include <iostream>
# include "Stiva.h"

using namespace std;

Stiva::Stiva(int dim) {
	this->dim = dim;
	crt = 0;
	v = new int[dim];
}

void Stiva::push(int x) {
	if (crt < dim) {
		v[crt++] = x;
	}
}


int Stiva::pop() {
	if (crt) {
		return v[--crt];
	} else {
		return -1;
	}
}

void Stiva::afisare() {
	for (int i = 0; i < crt; i++) {
		cout << v[i] << endl;
	}
}

# include <iostream>
# include "Stiva.h"

using namespace std;

int main() {
    Stiva s(4);
	Stiva t(s);
	
	s.push(1);
	s.push(2);
	t.push(10);
	t.push(20);
	
	cout << s.pop() << endl;
	cout << t.pop();
	
	return 0;
}
