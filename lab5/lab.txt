#include <stdio.h>

int sumad (int n)
{
   int s, d;
   s = n + 1;

   for(d = 2; d <= n/2; d++)
	   if (n%d == 0) s = s + d;
   
   return s;
}

		}

		for (j = 0; j < n; j++){

		}
		
		if(maxl < 0) maxl = -1;

	}
	
	for (i = 0; i < n; i++) {
		v2[i] = v[i][i];
	}
	
	for (i = 0; i < n; i++){
		 for (j = i + 1; j < n; j++){

				v2[j] = aux;
			 }
		 }
	}
